Creational Patterns

- to create objects without adding extra dependencies to the program

Factory

- Used to prevent tight coupling of modules.

Code without factory

								//pizza.js
								class Pizza {
									constructor(name) {
										this.name = name;
									}
								}
								module.exports = Pizza;
							
								
									//index.js
									const Pizza = require('/pizza');

									const eat = new Pizza('peprony');

index.js module depends on pizza.js module

Solution

Move the creation of objects to another class

							
								//pizza.js
								class Pizza {
									constructor(name) {
										this.name = name;
									}
								}
								module.exports = Pizza;

								//burger.js
								class Burger {
									constructor(name) {
									this.name = name;
									}
								}
							
								module.exports = Burger;						   

								//factory.js
								const Burger = require('./burger.js');
								const Pizza = require('./pizza.js');
								
								class Factory {
									static createEat(type, ...args) {
									switch (type) {
										case 'pizza': 
										return new Pizza(...args);
										case 'burger': 
										return new Burger(...args);
										}
									}
								}
								
								module.exports = Factory;
								
								// index.js
								const Factory = require('./factory.js');

								const eat = Factory.createEat('burger','shefburger')

advantages and disadvantages

+ eliminate the binding of one class to another classes

+ the whole logic of type creation is concentrated in one place

- increasement of complexity with a small number of classes

builder.js 
              
                class Pizza {
                  constructor(builder) {
                    this.cheese = builder.cheese;
                    this.sausage = builder.sausage;
                    this.tomatoes = builder.tomatoes;
                    this.onion = builder.onion;
                    this.sauce = builder.sauce;
                  }
                }
                
                class PizzaBuilder {
                  constructor() {
                    this.cheese = null;
                    this.sausage = null;
                    this.tomatoes = null;
                    this.onion = null;
                    this.sauce = null;
                  }
                
                  setCheese(cheese) {
                    this.cheese = cheese;
                    return this;
                  }
                
                  setSausage(sausage) {
                    this.sausage = sausage;
                    return this;
                  }
                  
                  setTomatoes(tomatoes) {
                    this.tomatoes = tomatoes;
                    return this;
                  }
                  
                  setOnion(onion) {
                    this.onion = onion;
                    return this;
                  }
                  
                  setSauce(sauce) {
                    this.sauce = sauce;
                    return this;
                  }
                
                  buildPizza() {
                    return new Pizza(this);
                  }
                  
                }
                
                class Director {
                  constructor(builder) {
                    this.builder = builder;
                  }
                  tastyPizza () {
                    this.builder
                      .setCheese(new Cheese())
                      .setOnion(new Onion())
                      .setSauce(new Sauce())
                      .setSausage(new Sausage())
                      .setTomatoes(new Tomatoes())
                    return this.builder.buildPizza();
                  }
              
                  studentPizza () {
                    this.builder
                      .setCheese(new Cheese())
                      .setSauce(new Sauce())
                      .setSausage(new Sausage())
                    return this.builder.buildPizza();
                  }
              
                }

Adapter

- Used when we want to add a class, but its interface that we want to use does not match the client code.

Adapter

              
								
								/*
								Lion interface :
								roar()
								*/

								class AfricanLion  {
									roar() {}
								}

								class AsianLion  {
									roar() {}
								}

Adapter

							
								class Hunter {
									hunt(lion) {
										// ... some code before
										lion.roar()
										//... some code after
									}
								}

Adapter

							
								// This needs to be added to the game
								class WildDog {
									bark() {}
								}
								
								// Adapter around wild dog to make it compatible with our game
								class WildDogAdapter {
									constructor(dog) {
										this.dog = dog;
									}
									
									roar() {
										this.dog.bark();
									}
								}

Now our dog can roar =)

Adapter

							
								wildDog = new WildDog()
								wildDogAdapter = new WildDogAdapter(wildDog)
								
								hunter = new Hunter()
								hunter.hunt(wildDogAdapter)

advantages and disadvantages

+ separates and hides from the client the details of converting various interfaces

+ This is the best way to adapt old code to new

- complicates code due to the addition of new classes

Patterns of Behavior

- connected with the way that objects communicate with each other

Observer

- Defines a dependency between objects so that whenever an object changes its state, all its changes are notified.

Observer

              		
								//jobseeker.js				
								class JobSeeker {
									constructor(name) {
										this._name = name
									}
							
									notify(jobPost) {
										document.write(this._name, 'has been notified of a new posting :', jobPost.title)
									}
								}

Observer

							
								//jobboard.js
								class JobBoard {
									constructor() {
										this._subscribers = []
									}
							
									subscribe(jobSeeker) {
										this._subscribers.push(jobSeeker)
									}

									unSubscribe(jobSeeker) {
										this._subscribers.forEach((item, index) => {
											item === jobSeeker ? this._subscribers.splice(index, 1) : false;
										})
									}
							
									addJob(jobPosting) {
										this._subscribers.forEach(subscriber => {
											subscriber.notify(jobPosting)
										})
									}
								}

Observer

							
								//index.js
								const JobPost = title => ({
									title: title;

								// Add a new job and see if subscribers get notified
