Factory

to create objects without adding extra dependencies to the program

Code without factory

								//pizza.js
								class Pizza {
									constructor(name) {
										this.name = name;
									}
								}
								module.exports = Pizza;
							
								
									//index.js
									const Pizza = require('/pizza');

									const eat = new Pizza('peprony');

index.js module depends on pizza.js module

Solution

Move the creation of objects to another class

							
								//pizza.js
								class Pizza {
									constructor(name) {
										this.name = name;
									}
								}
								module.exports = Pizza;

								//burger.js
								class Burger {
									constructor(name) {
									this.name = name;
									console.log(name); // name of burger
									}
								}
							
								module.exports = Burger;						   

								//factory.js
								const Burger = require('./burger.js');
								const Pizza = require('./pizza.js');
								
								class Factory {
									static createEat(type, ...args) {
									switch (type) {
										case 'pizza': 
										return new Pizza(...args);
										case 'burger': 
										return new Burger(...args);
										}
									}
								}
								
								module.exports = Factory;
                // index.js
								const Factory = require('./factory.js');

								const eat = Factory.createEat('burger','shefburger')

advantages and disadvantages

+ ridding the class of binding to other classes

+ the whole logic of type creation is concentrated in one place

- increase in complexity with a small number of classes

builder.js 
              
                class Pizza {
                  constructor(builder) {
                    this.cheese = builder.cheese;
                    this.sausage = builder.sausage;
                    this.tomatoes = builder.tomatoes;
                    this.onion = builder.onion;
                    this.sauce = builder.sauce;
                  }
                }
                
                class PizzaBuilder {
                  constructor() {
                    this.cheese = null;
                    this.sausage = null;
                    this.tomatoes = null;
                    this.onion = null;
                    this.sauce = null;
                  }
                
                  setCheese(cheese) {
                    this.cheese = cheese;
                    return this;
                  }
                
                  setSausage(sausage) {
                    this.sausage = sausage;
                    return this;
                  }
                  
                  setTomatoes(tomatoes) {
                    this.tomatoes = tomatoes;
                    return this;
                  }
                  
                  setOnion(onion) {
                    this.onion = onion;
                    return this;
                  }
                  
                  setSauce(sauce) {
                    this.sauce = sauce;
                    return this;
                  }
                
                  buildPizza() {
                    return new Pizza(this);
                  }
                  
                }
                
                class Director {
                  constructor(builder) {
                    this.builder = builder;
                  }
                  tastyPizza () {
                    this.builder
                      .setCheese(new Cheese())
                      .setOnion(new Onion())
                      .setSauce(new Sauce())
                      .setSausage(new Sausage())
                      .setTomatoes(new Tomatoes())
                    return this.builder.buildPizza();
                  }
              
                  studentPizza () {
                    this.builder
                      .setCheese(new Cheese())
                      .setSauce(new Sauce())
                      .setSausage(new Sausage())
                    return this.builder.buildPizza();
                  }
              
                }
